    is_md5_equal.php

    <?php 
    function readmyfile($path){
        $fh = fopen($path, "rb");
        $data = fread($fh, filesize($path));
        fclose($fh);
        return $data;
    }
    $a = urlencode(readmyfile('D:\Code\MD5_collision\fastcoll\msg_msg1.txt'));
    $b = urlencode(readmyfile('D:\Code\MD5_collision\fastcoll\msg_msg2.txt'));
    if(md5((string)urldecode($a))===md5((string)urldecode($b))){
        echo $a."\n\n";
    }
    if(urldecode($a)!=urldecode($b)){
        echo $b;
    }
